## Install

```bash
npm install --save componentegithub
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'componentegithub'
import 'componentegithub/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
